  <!-- ‚úÖ Script to connect with backend -->
  <script>
    async function generateSummary() {
      const text = document.getElementById("userInput").value.trim();
      const outputDiv = document.getElementById("output");

      if (!text) {
        outputDiv.innerHTML = "<p style='color:red'>‚ö†Ô∏è Please enter some text.</p>";
        return;
      }

      outputDiv.innerHTML = "<p>‚è≥ Generating summary...</p>";

      try {
        const response = await fetch("https://turbolearn-backend.onrender.com/generate-summary", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ text }),
        });

        const data = await response.json();
        outputDiv.innerHTML = `<h3>üìå Summary:</h3><p>${data.summary}</p>`;
      } catch (error) {
        console.error(error);
        outputDiv.innerHTML = "<p style='color:red'>‚ùå Error: Failed to fetch summary.</p>";
      }
    }
  </script>
